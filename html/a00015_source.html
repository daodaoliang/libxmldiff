65 /** Populate item list with id strings of child nodes.
66  *
67  * This list will be populated with the name of the node and
68  * attributes/elements contained in the ids list, and then sorted.

70  * @param node : the node to inspect
71  * @param itemlist : the list to populate
72  * @param options : diff options

74 void populate_itemlist(xmlNodePtr node, vector<idNode> & itemlist, const struct xmldiff_options & options)
75 {
76  xmlNodePtr curNode, tmpNode;
77  idNode curNodeIdent;
78  vector<xmlstring>::const_iterator id;
79 
80  curNode = node->children;
81  while (curNode != NULL)
82  {
83  if (!matchNode(curNode, options.ignore))
84  {
85  switch(curNode->type)
86  {
87  case XML_ELEMENT_NODE:
88  curNodeIdent.id = (((curNode->ns) && (curNode->ns->prefix))?curNode->ns->prefix:BAD_CAST"") + xmlstring(BAD_CAST ":") + curNode->name;
89  for (id = options.ids.begin(); id != options.ids.end(); id++)
90  {
91  if (id->at(0) == '@')
92  {
93  if (xmlHasProp(curNode, id->c_str()+1))
94  {
95  curNodeIdent.id += strSpace + *id + strEqual + strQuote + (xmlChar *)xmlCharTmp(xmlGetProp(curNode, id->c_str()+1)) + strQuote;
96  }
97  }
98  else if (id->at(0) == '.')
99  {
100  curNodeIdent.id += strSpace + strEqual + getNodeTextOnly(curNode);
101  }
102  else
103  {
104  tmpNode = getFirstChildByTagName(curNode, id->c_str());
105  if (tmpNode != NULL)
106  {
107  curNodeIdent.id += strSpace + *id + strEqual + (xmlChar *)xmlCharTmp(xmlNodeGetContent(tmpNode));
108  }
109  }
110  }
111  curNodeIdent.node = curNode;
112  itemlist.push_back(curNodeIdent);
113  break;
114  // Ignore Text and Attributes
115  case XML_ATTRIBUTE_NODE :
116  case XML_TEXT_NODE :
117  break;
118  // Text-Style Content
119  case XML_PI_NODE:
120  case XML_COMMENT_NODE:
121  case XML_CDATA_SECTION_NODE :
122  switch(curNode->type)
123  {
124  case XML_PI_NODE: curNodeIdent.id = xmlstring(BAD_CAST "<?") + curNode->name; break;
125  case XML_COMMENT_NODE: curNodeIdent.id = BAD_CAST "<!--"; break;
126  case XML_CDATA_SECTION_NODE: curNodeIdent.id = BAD_CAST "#CDATA"; break;
127  default: curNodeIdent.id = ((curNode->ns)?curNode->ns->prefix:BAD_CAST"") + xmlstring(BAD_CAST ":") + curNode->name; break;
128  }
129  // TODO : Use hash code instead
130  // curNodeIdent.id += strSpace + strEqual + getNodeTextOnly(curNode);
131  curNodeIdent.id += strSpace + strEqual + curNode->content;
132  curNodeIdent.node = curNode;
133  itemlist.push_back(curNodeIdent);
134  break;
135  // Ignored items
136  case XML_NAMESPACE_DECL :
137  case XML_DOCUMENT_NODE :
138  case XML_DOCUMENT_TYPE_NODE :
139  case XML_DOCUMENT_FRAG_NODE :
140  case XML_HTML_DOCUMENT_NODE :
141  break;
142  default:
143  // TODO : Should emit something like a warning.

145  break;
146  }
147  }

153 /** Main diff function.
154  *
155  * This function performs the diff between two nodes, and modify
156  * the nodeAfter to reflect the changes (diff:status, add elements,...)
157  * It also free nodeBefore as much as possible, to save memory.
158  * The algorithm is described in the function.
159  *
160  * @param nodeBefore : the node of the first file
161  * @param nodeAfter : the node of the second file
162  * @param options : the diff options (see xmldiff_options)
163  * @return the status of the diff (@see DN_STATUS) or error code :
164  * -1 : unable to create some structures, should be a memory problem.
165  *
166  *
167  * @warning nodeBefore and nodeAfter are affected by this function.
168  */

173  int status;

175  xmlNodePtr insertPoint;
176  //xmlNsPtr curNs;
177  vector<idNode> * listBefore;
178  vector<idNode> * listAfter;
179  vector<idNode>::iterator iterBefore;
180  vector<idNode>::iterator iterAfter;
181  vector<xmlNodePtr> removed_nodes;

183  vector<xmlNodePtr>::iterator iter;
184 
185  xmlstring s; // char ss[100];
186

188  // Compare elements
189  // * Create lists
190  listBefore = new vector<idNode>;
191  listAfter = new vector<idNode>;
192  if ((listBefore == NULL) || (listAfter == NULL)) return -1;
193  // * Populate lists
194  populate_itemlist(nodeBefore, *listBefore, options);
195  populate_itemlist(nodeAfter, *listAfter, options);
196  // * Iterate lists
197  iterBefore = listBefore->begin();
198  iterAfter = listAfter->begin();
199  while ((iterBefore != listBefore->end()) || (iterAfter != listAfter->end()))
200  {
201  // Compare the two current id strings.
202  if (iterBefore == listBefore->end()) cmp = -1;
203  else if (iterAfter == listAfter->end()) cmp = 1;
204  else cmp = iterAfter->id.compare(iterBefore->id);
205  switch(cmp)
206  {
207  case -1:
208  // If the id string after < before, the element has been added
209  if (!(options.diffOnly || options.doNotTagDiff))
210  setAttributeToAllChilds(
211  iterAfter->node,
212  options.diff_xmlns,
213  options.diff_attr,
214  options.diffQualifiersList[DN_ADDED],
215  options.tagChildsAddedRemoved);
216  iterAfter++; afterNodesCur++; nodesCur++;
217  status = DN_BELOW;
218  break;
219  case 1:
220  // If the id string after > before, the element has been removed
221  // As we cannot insert it now, we add it in the removed_nodes list
222  // which will be processed later.
223  removed_nodes.push_back(iterBefore->node);
224  // diff:status attribute is set later, as the copy in new document cause problems with namespaces
225  iterBefore++; beforeNodesCur++; nodesCur++;
226  status = DN_BELOW;
227  break;
228  default:
229  // If the two id are equal, these nodes are comparables
230  // Due to memory reason (should free the id lists before recursing...)
231  // these items will be placed in a list to be processed later.
232  // Also the correspondance between these nodes is stored as UserData
233  iterAfter->node->_private = (void *) iterBefore->node;
234  iterBefore->node->_private = (void *) iterAfter->node;
235  subnodes_to_diff.push_back(iterBefore->node);
236  iterAfter++; afterNodesCur++; nodesCur++;
237  iterBefore++; beforeNodesCur++;
238  break;
239  }
240  }
241  // * Free lists
242  if (listBefore) delete listBefore;
243  if (listAfter) delete listAfter;
244  // * Add Removed nodes to current tree
245  if (!options.diffOnly)
246  {
247  for (iter = removed_nodes.begin();
248  iter != removed_nodes.end();)
249  {
250  insertPoint = NULL;
251  curNode = *iter;
252  iter++;
253  tmpNode = curNode;
254  // Unlink node
255  bisNode = curNode;
256  if (options.doNotFreeBeforeTreeItems)
257  {
258  bisNode = xmlDocCopyNode(curNode, nodeAfter->doc, 1);
259  }
260  xmlUnlinkNode(bisNode);
261 
262  // Search insert point
263  // - add before the next pair found
264  while (tmpNode != NULL)
265  {
266  insertPoint = (xmlNodePtr) tmpNode->_private;
267  if (insertPoint != NULL) tmpNode = NULL;
268  else tmpNode = tmpNode->next;
269  }
270  if (insertPoint != NULL)
271  {
272  xmlAddPrevSibling(insertPoint, bisNode);
273  }
274  else
275  {
276  // - if we have not found, add after the previous pair found
277  tmpNode = curNode;
278  while (tmpNode != NULL)
279  {
280  insertPoint = (xmlNodePtr) tmpNode->_private;
281  if (insertPoint != NULL) tmpNode = NULL;
282  else tmpNode = tmpNode->prev;
283  }
284  if (insertPoint != NULL)
285  {
286  xmlAddNextSibling(insertPoint, bisNode);
287  }
288  else
289  {
290  // - not found ? add at the end of the child list
291  xmlAddChild(nodeAfter, bisNode);
292  }
293  }
294  if (!options.doNotTagDiff)
295  setAttributeToAllChilds(bisNode,
296  options.diff_xmlns,
297  options.diff_attr,
298  options.diffQualifiersList[DN_REMOVED],
299  options.tagChildsAddedRemoved);
300  if ((!options.diffOnly) && (bisNode->type == XML_ELEMENT_NODE))
301  if (xmlReconciliateNs(nodeAfter->doc, bisNode) < 0) throwError(XD_Exception::XDE_NAMESPACE_PROBLEM, "Unable to reconciliate Namespaces");
302  }
303  }
304  removed_nodes.clear();
305  // * Diff Sub-Nodes
306  for (iter = subnodes_to_diff.begin();
307  iter != subnodes_to_diff.end();
308  iter++)
309  {
310  xmlNodePtr iA, iB;
311  iB = *iter;
312  iA = (xmlNodePtr) iB->_private;
313  if (diffNode(iB, iA, options) != DN_NONE) status = DN_BELOW;
314  if ((!options.diffOnly) && (!options.doNotFreeBeforeTreeItems))
315  {
316  xmlUnlinkNode(iB);
317  xmlFreeNode(iB);
318  }
319  }
320  subnodes_to_diff.clear();
321  // Compare attributes
322  if (nodeAfter->type == XML_ELEMENT_NODE)
323  {
324  xmlAttrPtr curAttr, attAfter, attBefore;
325  xmlstring xAttBefore, xAttAfter;
326  // Look for modified/added attributes
327  for (curAttr = nodeAfter->properties; curAttr != NULL; curAttr = curAttr->next)
328  {

330  {
331  xAttBefore = BAD_CAST "";

333  {
334  xAttBefore = xmlCharTmp(xmlNodeListGetString(nodeBefore->doc, attBefore->children, 1));
335  }
336  xAttAfter = xmlCharTmp(xmlNodeListGetString(nodeAfter->doc, curAttr->children, 1));
337  if (xAttAfter.compare(xAttBefore) != 0)
338  {
339  if ((!options.diffOnly) && ((options.beforeValue) || (xAttAfter.compare(BAD_CAST "") == 0)))
340  {
341  s = xAttBefore;
342  s += options.separator;
343  s += xAttAfter;
344  xmlSetNsProp(nodeAfter, curAttr->ns, curAttr->name, s.c_str());
345  }
346  status = DN_MODIFIED;
347  }
348  }
349  }
350  // Removed attributes ?
351  for (curAttr = nodeBefore->properties; curAttr != NULL; curAttr = curAttr->next)
352  {
353  if (!matchNode((xmlNodePtr)curAttr, options.ignore))
354  {
355  attAfter = xmlHasNsProp(nodeAfter, curAttr->name, (curAttr->ns)?curAttr->ns->href:NULL);
356  if (attAfter == NULL)
357  {
358  if (!options.diffOnly)
359  {
360  s = xmlCharTmp(xmlNodeListGetString(nodeBefore->doc, curAttr->children, 1));
361  s += options.separator;
362  if (curAttr->ns != NULL)
363  {
364  if (xmlSearchNsByHref(nodeAfter->doc, nodeAfter, curAttr->ns->href) == NULL)
365  {
366  xmlSetNsProp(nodeAfter, curAttr->ns, curAttr->name, s.c_str());
367  if (xmlReconciliateNs(nodeAfter->doc, nodeAfter) < 0) throwError(XD_Exception::XDE_NAMESPACE_PROBLEM, "Unable to reconciliate Namespaces");
368  }
369  else
370  {
371  xmlSetNsProp(nodeAfter, xmlSearchNsByHref(nodeAfter->doc, nodeAfter, curAttr->ns->href), curAttr->name, s.c_str());
372  }
373  }
374  else
375  {
376  xmlSetNsProp(nodeAfter, NULL, curAttr->name, s.c_str());
377  }
378  }
379  status = DN_MODIFIED;
380  }
381  }
382  }
383  }
384  // Compare values
385  xmlstring valBefore, valAfter;
386  valBefore = getNodeTextOnly(nodeBefore);
387  valAfter = getNodeTextOnly(nodeAfter);
388  if (valBefore.compare(valAfter))
389  {
390  if ((!options.diffOnly) && ((options.beforeValue) || (valAfter.compare(BAD_CAST "") == 0)))
391  {
392  s = valBefore;
393  s += options.separator;
394  if (nodeAfter->children == NULL)
395  {
396  xmlAddChild(nodeAfter, xmlNewText(s.c_str()));
397  }
398  else
399  {
400  xmlAddPrevSibling(nodeAfter->children, xmlNewText(s.c_str()));
401  }
402  }
403  status = DN_MODIFIED;
404  }
405  // Update Progess Status
406  if ( (afterNodesNb != 0) || (beforeNodesNb != 0))
407  {
408  curPercent = (int)(((afterNodesCur + beforeNodesCur) * 100) / (afterNodesNb + beforeNodesNb));
409  if (curPercent != precPercent)
410  {
411  if (options.callbackProgressionPercent != NULL)
412  {
413  options.callbackProgressionPercent(curPercent, precPercent, beforeNodesNb, afterNodesNb, nodesCur, options.cbProgressionArg);
414  }
415  precPercent = curPercent;
416  }
417  }
418  // Update Status if Element
419  if ((!(options.diffOnly || options.doNotTagDiff)) && (status != DN_NONE) && (nodeAfter->type == XML_ELEMENT_NODE))
420  xmlSetNsProp(nodeAfter,
421  xmlSearchNs(nodeAfter->doc, nodeAfter, options.diff_xmlns.c_str()),
422  options.diff_attr.c_str(),
423  options.diffQualifiersList[status].c_str());
424  // Delete the node if we must keep only differences
425  if ((!options.diffOnly) && (status == DN_NONE) && (options.keepDiffOnly))
426  {
427  // FIXME : Quick Hack to void crash under Linux
428  if (nodeAfter->type != XML_DOCUMENT_NODE)
429  {
430  xmlUnlinkNode(nodeAfter);
431  xmlFreeNode(nodeAfter);
432  }
433  }
434  return status;

437 int diffTree(xmlNodePtr nodeBefore, xmlNodePtr nodeAfter, const struct xmldiff_options & options)
438 {
439  int status;
440  xmlNodePtr curNode;
441  xmlNsPtr curNs;
442  beforeNodesNb = 0; afterNodesNb = 0;
443  if (options.callbackProgressionPercent != NULL)
444  {
445  // Progression is wanted, count number of element used
446  beforeNodesCur = 0; afterNodesCur = 0; nodesCur = 0; precPercent = 0;
447  beforeNodesNb = countElementNodes(nodeBefore);
448  options.callbackProgressionPercent(-1, -1, beforeNodesNb, 0, 0, options.cbProgressionArg);
449  afterNodesNb = countElementNodes(nodeAfter);
450  options.callbackProgressionPercent(-2, -2, beforeNodesNb, afterNodesNb, 0, options.cbProgressionArg);
451  }
452  // Add the namespace
453  if (!options.diffOnly)
454  {
455  if ((options.diff_ns.length() > 0) && (options.diff_xmlns.length() > 0))
456  {
457  createNamespaceOnTop(nodeAfter, options.diff_ns.c_str(), options.diff_xmlns.c_str());
458  }
459  if (options.mergeNsOnTop)
460  {
461  // Search NS Definition node on before file
462  if (nodeBefore->type == XML_DOCUMENT_NODE) curNode = nodeBefore->children; else curNode = nodeBefore;
463  while ((curNode != NULL) && (curNode->type != XML_ELEMENT_NODE)) curNode = curNode->next;
464  if (curNode) curNs = curNode->nsDef;
465  // Search NS Definition node on after file
466  if (nodeAfter->type == XML_DOCUMENT_NODE) curNode = nodeAfter->children; else curNode = nodeAfter;
467  while ((curNode != NULL) && (curNode->type != XML_ELEMENT_NODE)) curNode = curNode->next;
468  while (curNs != NULL)
469  {
470  if (!xmlSearchNsByHref(curNode->doc, curNode, curNs->href))
471  {
472  if (xmlNewNs(curNode, curNs->href, curNs->prefix) == NULL)
473  throwError(XD_Exception::XDE_NAMESPACE_PROBLEM, "Error merging the Namespace %s (%s)", curNs->href, curNs->prefix);
474  }
475  curNs = curNs->next;
476  }
477  }
478  }
479  status = diffNode(nodeBefore, nodeAfter, options);
480  if (options.callbackProgressionPercent != NULL)
481  {
482  options.callbackProgressionPercent(100, precPercent, beforeNodesNb, afterNodesNb, nodesCur, options.cbProgressionArg); precPercent = 100;
483  options.callbackProgressionPercent(-3, -3, beforeNodesNb, afterNodesNb, nodesCur, options.cbProgressionArg);
484  }
485  return status;

490 {
491  xmlNodePtr curNode;
492  xmlAttrPtr curProp;
493  xmlstring strStatus;
494  int status;
495  status = DN_NONE;
496  if (node->type == XML_ELEMENT_NODE)
497  {
498  // Get Current Status
499  strStatus = BAD_CAST "";
500  if (xmlHasNsProp(node, options.diff_attr.c_str(), options.diff_ns.c_str()))
501  strStatus = xmlstring(xmlGetNsProp(node, options.diff_attr.c_str(), options.diff_ns.c_str()));
502  status = DN_NONE;
503  if (strStatus == options.diffQualifiersList[DN_ADDED]) status = DN_ADDED;
504  else if (strStatus == options.diffQualifiersList[DN_REMOVED]) status = DN_REMOVED;
505  else if (strStatus == options.diffQualifiersList[DN_MODIFIED]) status = DN_MODIFIED;
506  else if (strStatus == options.diffQualifiersList[DN_BELOW]) status = DN_BELOW;
507  else if (strStatus == options.diffQualifiersList[DN_NONE]) status = DN_NONE;
508  strStatus = BAD_CAST "";
509  // Force BELOW to NONE to Recalculate
510  if (status == DN_BELOW) status = DN_NONE;
511  // If MODIFIED, check that the modified item is here
512  // (not verified if beforeValue not set
513  if ((status == DN_MODIFIED) && (options.beforeValue)) status = DN_NONE;
514  }
515  // Childrens
516  curNode = node->children;
517  while (curNode != NULL)
518  {
519  switch(curNode->type)
520  {
521  case XML_ELEMENT_NODE:
522  if (recalcTree(curNode, options) != DN_NONE)
523  {
524  if (status == DN_NONE) status = DN_BELOW;
525  }
526  break;
527  case XML_TEXT_NODE:
528  if (xmlStrstr(curNode->content, options.separator.c_str()) != NULL)
529  status = DN_MODIFIED;
530  break;
531  default:
532  break;
533  }
534  curNode = curNode->next;
535  }
536  // Properties
537  curProp = node->properties;
538  while (curProp != NULL)
539  {
540  switch(curProp->type)
541  {
542  case XML_ATTRIBUTE_NODE:
543  if (xmlStrstr(xmlCharTmp(xmlNodeListGetString(curProp->doc, curProp->children, 1)),
544  options.separator.c_str()) != NULL)
545  status = DN_MODIFIED;
546  break;
547  default:
548  break;
549  }
550  curProp = curProp->next;
551  }
552  // Update Status
553  if ((node->type == XML_ELEMENT_NODE) && (!options.diffOnly))
554  {
555  if (status != DN_NONE)
556  {
557  xmlSetNsProp(node,
558  xmlSearchNs(node->doc, node, options.diff_xmlns.c_str()),
559  options.diff_attr.c_str(),
560  options.diffQualifiersList[status].c_str());
561  }
562  else
563  {
564  xmlUnsetNsProp(node,
565  xmlSearchNs(node->doc, node, options.diff_xmlns.c_str()),
566  options.diff_attr.c_str());
567  }
568  }
