77  bool beforeValue;
78 
79  /** If set, the diff algorithm will not try to optimize memory while diffing.
80  * If not set, the diff will free each processed node of the before tree.

82  * Useless if diffOnly is true.
83  */
84  bool doNotFreeBeforeTreeItems;
85 
86  /** With this option, trees wont be affected by the diff.
87  * If you only want to reuse trees, consider duplicating the after node,

89  * If your goal is only to have the status modified / none with root nodes

92  bool diffOnly;

99  bool keepDiffOnly;

158  /** Percent Progression Callback
159  * If defined, this callback will be called each percent.
160  * The application can define display, or other user information (memory,...)
161  *
162  * Set this to NULL if you do not use this (will speed up as no node counting will be done)
163  *
164  * This function will be called :
165  * - one time after counting elements of before tree (percent = -1)
166  * - one time after counting elements of after tree (percent = -2)
167  * - each time the percent have changed : on a tree with four nodes,
168  * processing a node will lead to 25% progression, but the function
169  * will be called only once, not 25 times.
170  *
171  * @param percent Progression Percent
172  * (estimated with (nbNodesBefore + nbNodesAfter) * 100 / (2 * nbNodesProcessed) )
173  * @param prec Last progression percent
174  * @param nbNodesBefore Number of nodes in the before tree
175  * @param nbNodesAfter Number of nodes in the after tree
176  * @param nbNodesProcessed Number of processed nodes (due to added/removed nodes,
177  * this can be larger than max(nbNodesBefore, nbNodesAfter).
178  *
179  * @warning The application *must* not call a second time the diff function. No
180  * check will be done. Behaviour is not guaranteed.
181  * @warning UNIMPLEMENTED
182  */
183  void (* callbackProgressionPercent)(int percent, int prec,
184  long nbNodesBefore,
185  long nbNodesAfter,
186  long nbNodesProcessed,
187  void * arg);
