281 /// Diff Trees XML Files
282 int diffXmlFiles(string beforeAlias, string afterAlias, string outputAlias, const struct globalOptions & options)
283 {
284  int rc;
285  xmlNodePtr beforeNode, afterNode, outputNode;
286 
287  // Prepare the nodes
288  beforeNode = getXmlFile(beforeAlias, options);
289  afterNode = getXmlFile(afterAlias, options);
290  if (options.diffOnly)
291  {
292  // Set output to after node.
293  // No result is waited, so there is no need to create a file
294  outputNode = afterNode;
295  }
296  else if (options.optimizeMemory)
297  {
298  // Transform after into output node
299  outputNode = afterNode;
300  loadedFiles[outputAlias] = loadedFiles[afterAlias];
301  loadedFiles[outputAlias].filename = outputAlias;
302  loadedFiles[outputAlias].modified = true;
303 #ifndef WITHOUT_LIBXSLT
304  loadedFiles[outputAlias].xslt = NULL;
305 #endif
306  loadedFiles.erase(afterAlias);
307  }
308  else
309  {
310  // We must duplicate the after tree and insert it in modified state
311  // Tricky thing :
312  // - we backup the document of the current node, and assign it back to afterAlias
313  // - the original selected node becomes outputNode
314  // => we do keep the same selected node
315  outputNode = (xmlNodePtr)xmlCopyDoc(afterNode->doc, 1);
316  loadedFiles[afterAlias].doc = (xmlDocPtr)outputNode;
317  if (outputNode == NULL) throwError(XD_Exception::XDE_DIFF_MEMORY_ERROR, "Unable to instanciate stuctures, probably due to a memory problem");
318  loadedFiles[outputAlias] = loadedFiles[afterAlias];
319  loadedFiles[outputAlias].doc = afterNode->doc;
320  loadedFiles[outputAlias].filename = outputAlias;
321  loadedFiles[outputAlias].modified = true;
322 #ifndef WITHOUT_LIBXSLT
323  loadedFiles[outputAlias].xslt = NULL;
324 #endif
325  outputNode = afterNode;
326  }
327 
328  struct xmldiff_options opt;
329  opt = options;
330  opt.doNotFreeBeforeTreeItems = !options.optimizeMemory;
331  if (options.verboseLevel < 2) opt.callbackProgressionPercent = NULL;
332 
333  // Perform the diff
334  rc = diffTree(beforeNode, outputNode, opt);
335  if (rc == -1) throwError(XD_Exception::XDE_DIFF_MEMORY_ERROR, "Unable to instanciate stuctures, probably due to a memory problem");
336  if (rc == DN_NONE) verbose(3, options.verboseLevel, "Files are similar.\n");
337  else verbose(3, options.verboseLevel, "Files are different.\n");
338 
339  // Do not write the file if not necessary
340  if ((rc == DN_NONE) && (options.optimizeMemory || options.keepDiffOnly) ) loadedFiles[outputAlias].modified = false;
341 
342  // Cleaning stuff
343  if ((!options.diffOnly) && (options.optimizeMemory))
344  {
345  closeXmlFile(beforeAlias, options);
346  }
347  return rc;
348 }

366 /// Delete nodes of the file
367 int deleteNodes(const string & alias, const xmlstring & xpath, const struct globalOptions & options)
368 {
369  int i, nb;
370  xmlNodePtr node, curNode;
371  xmlXPathContextPtr xpathCtx;
372  xmlXPathObjectPtr xpathObj;
373  xmlNsPtr ns;
374 
375  node = getXmlFile(alias, options); if (node == NULL) return -1;
376  xpathCtx = xmlXPathNewContext(node->doc);
377  if(xpathCtx == NULL) throwError(XD_Exception::XDE_DIFF_MEMORY_ERROR, "Unable to instanciate stuctures, probably due to a memory problem");
378 
379  // Populate NS with the first top-level element
380  curNode = NULL;
381  if (node->doc != NULL) curNode = node->doc->children;
382  while ((curNode != NULL) && ((curNode->type != XML_ELEMENT_NODE))) curNode = curNode->next;
383  if (curNode != NULL)
384  {
385  xpathCtx->namespaces = xmlGetNsList(node->doc, curNode);
386  xpathCtx->nsNr = 0;
387  if (xpathCtx->namespaces) ns = *xpathCtx->namespaces; else ns = NULL;
388  while (ns != NULL) { xpathCtx->nsNr++; ns = ns->next; }
389  }
390 
391  xpathObj = xmlXPathEvalExpression(xpath.c_str(), xpathCtx);
392  if (xpathObj != NULL)
393  {
394  nb = ((xpathObj->nodesetval)?xpathObj->nodesetval->nodeNr:0);
395  verbose(2, options.verboseLevel, "Deleting %s = %d nodes...", BAD_CAST xpath.c_str(), nb);
396  for(i=0;i<nb;i++)
397  {
398  xmlUnlinkNode(xpathObj->nodesetval->nodeTab[i]);
399  xmlFreeNode(xpathObj->nodesetval->nodeTab[i]);
400  }
401  verbose(2, options.verboseLevel, " done.\n");
402  xmlXPathFreeObject(xpathObj);
403  if (nb != 0) loadedFiles[alias].modified = true;
404  }
405  xmlXPathFreeContext(xpathCtx);
406  return 0;
407 }
