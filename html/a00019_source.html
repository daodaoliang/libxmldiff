285 int parseCommandLine(const vector<string> & cl, /* [in, out] */ struct appCommand & opt)
286 {
287  int curarg;
288  int nArgOther;
289  int ret;
290 
291  #define ARG_MATCH(x) (stricmp(cl[curarg].c_str(), x) == 0)
292 
293  nArgOther = 0;
294  // Parse command line
295  for (curarg = 0; curarg < cl.size(); curarg++)
296  {
297  if (ARG_MATCH("--help") || ARG_MATCH("-h") || ARG_MATCH("-?"))
298  {
299  opt.action = XD_HELP;
300  return 0;
301  }
302  else if (cl[curarg][0] == '-')
303  {
304  ret = parseOption(cl[curarg], (curarg + 1 >= cl.size())?"":cl[curarg+1], opt);
305  curarg += ret-1;
306  }
307  else
308  {
309  if (nArgOther == 0)
310  {
311  parseAction(cl[curarg], opt);
312  if (opt.action == XD_REM) return 0;
313  }
314  else if (nArgOther < LX_APPCOMMAND_NBPARAM)
315  {
316  opt.param[nArgOther-1] = cl[curarg];
317  }
318  else
319  {
320  throwError(XD_Exception::XDE_TOO_MANY_ARGUMENTS, "Too many arguments '%s'.", cl[curarg].c_str());
321  }
322  nArgOther++;
323  }
324  }
325  return 0;
326 }

330 {
331  int rc, nparam, ilen, olen, i;
332  map<string, string> vars;
333  charTmp tmp_params[2*LX_APPCOMMAND_NBPARAM];

335  char var[10];
336  xmlChar * temp;
337  struct appCommand cmd;
338 
339  rc = 0;
340  cmd = p_cmd;
341 
342  switch(cmd.action)
343  {
344  case XD_RECALC:
345  rc = recalcXmlFiles(cmd.param[0], cmd);
346  break;
347  case XD_LOAD:
348  rc = loadXmlFile(cmd.param[0], cmd.param[1], cmd);
349  break;
350  case XD_SAVE:
351  rc = saveXmlFile(cmd.param[0], cmd.param[1], cmd);
352  break;
353  case XD_CLOSE:
354  case XD_DISCARD:
355  rc = 0; closeXmlFile(cmd.param[0], cmd);
356  break;
357  case XD_FLUSH:
358  rc = 0; flushXmlFiles(cmd);
359  break;
360  case XD_EXECUTE:
361  vars["$$"] = "$";
362  for (i = 1; i < LX_APPCOMMAND_NBPARAM; i++)
363  {
364  if (cmd.param[i] != "")
365  {
366  if (i < 10) sprintf(var, "$%d",i); else sprintf(var, "$(%d)", i);
367  vars[var] = cmd.param[i];
368  }
369  }
370  rc = executeFile(cmd.param[0], vars, cmd);
371  break;
372  case XD_MERGE:
373  cmd.doNotTagDiff = true;
374  // continue XD_DIFF
375  case XD_DIFF:
376  rc = diffXmlFiles(cmd.param[0], cmd.param[1], cmd.param[2], cmd);
377  break;
378  case XD_PRINT:
379  rc = 0;
380  for(i = 0; i < LX_APPCOMMAND_NBPARAM; i++)
381  cout << cmd.param[i];
382  cout << endl;
383  break;
384  case XD_RET:
385  sscanf(cmd.param[0].c_str(),"%d", &i);
386  rc = i;
387  break;
388  case XD_DELETE:
389  rc = deleteNodes(cmd.param[0], BAD_CAST cmd.param[1].c_str(), cmd);
390  break;
391  case XD_DUP:
392  rc = duplicateDocument(cmd.param[0], cmd.param[1], cmd);
393  break;
394 #ifndef WITHOUT_LIBXSLT
395  case XD_XSLT:
396  nparam = 0;
397  for (i = 3; i < LX_APPCOMMAND_NBPARAM; i++)
398  {
399  if ((cmd.param[i] != "") && (cmd.param[i].find('=') != string::npos))
400  {
401  olen = 3 * cmd.param[i].length();
402  temp = (xmlChar *)malloc(olen + 1);
403  if (temp == NULL) { throwError(XD_Exception::XDE_MEMORY_ERROR, "Memory error while parsing.");}
404  ilen = cmd.param[i].substr(0, cmd.param[i].find('=')).length();
405  if (isolat1ToUTF8(temp, &olen, (const unsigned char *)cmd.param[i].substr(0, cmd.param[i].find('=')).c_str(), &ilen) == -1)
406  throwError(XD_Exception::XDE_OTHER_ERROR, "Error while converting input to UTF8.");
407  temp[olen] = 0;
408  tmp_params[nparam].setCharTmp(strdup((char *)temp));
409  params[nparam++] = (const char *)tmp_params[nparam-1];
410  olen = 3 * cmd.param[i].length();
411  ilen = cmd.param[i].substr(cmd.param[i].find('=') + 1, cmd.param[i].length()).length();
412  if (isolat1ToUTF8(temp, &olen, (const unsigned char *)cmd.param[i].substr(cmd.param[i].find('=') + 1, cmd.param[i].length()).c_str(), &ilen) == -1)
413  throwError(XD_Exception::XDE_OTHER_ERROR, "Error while converting input to UTF8.");
414  temp[olen] = 0;
415  tmp_params[nparam].setCharTmp((char *)temp);
416  params[nparam++] = (const char *)tmp_params[nparam-1];
417  verbose(5, cmd.verboseLevel, "Parameter \"%s\" = \"%s\"\n ", params[nparam-2], params[nparam-1]);
418  }
419  }
420  params[nparam] = NULL;
421  rc = applyStylesheet(cmd.param[0], cmd.param[1], cmd.param[2], params, cmd);
422  // while (nparam > 0) { free((void *)params[--nparam]); }
423  break;
424 #endif // WITHOUT_LIBXSLT
425  case XD_REM: break;
426  case XD_OPTIONS: break;
427  case XD_PRINTCONF:
428  printConfiguration(cmd);
429  break;
430  case XD_HELP:
431  usage();
432  break;
433  default:
434  throwError(XD_Exception::XDE_UNKNOWN_COMMAND, "Unknown command.");
435  break;

440 /** tokenizeCommand : parse command line
441  * @param command command string
442  * @return the list of tokens
443  */
444 vector<string> tokenizeCommand(string command)
445 {
446  string::const_iterator iter;
447  string buf = "";
448  enum eStatus {CL_NONE = 0, CL_IN_ARG, CL_IN_SIMPLE_QUOTE, CL_IN_DOUBLE_QUOTE} status = CL_NONE;
449  vector<string> cl;
450 
451  #define IS_BLANK(x) ((x == ' ') || (x == '\t') || (x == '\n') || (x == '\r'))
452 
453  for(iter = command.begin(); iter != command.end(); iter++)
454  {
455  switch(status)
456  {
457  case CL_NONE:
458  if (*iter == '"') status = CL_IN_DOUBLE_QUOTE;
459  else if (*iter == '\'') status = CL_IN_SIMPLE_QUOTE;
460  else if (!IS_BLANK(*iter)) { status = CL_IN_ARG; buf = *iter; }
461  break;
462  case CL_IN_ARG:
463  if IS_BLANK(*iter)
464  {
465  cl.push_back(buf); buf = "";
466  status = CL_NONE;
467  }
468  else buf += *iter;
469  break;
470  case CL_IN_DOUBLE_QUOTE:
471  case CL_IN_SIMPLE_QUOTE:
472  if ( ((*iter == '"') && (status == CL_IN_DOUBLE_QUOTE)) ||
473  ((*iter == '\'') && (status == CL_IN_SIMPLE_QUOTE)) )
474  {
475  cl.push_back(buf); buf = "";
476  status = CL_NONE;
477  }
478  else buf += *iter;
479  break;
480  }
481  }
482  if (buf != "") cl.push_back(buf); buf = "";
483  return cl;
484 }

488  * @param variables the list of variables name/value

492 {
493  int nb = 0, start = 0;
494  vector<string>::iterator iter;
495  map<string, string>::iterator var;
496  for(iter=tokens.begin(); iter != tokens.end(); iter++)
497  {
498  /* Old way : only replace full words
499  if (variables.find(*iter) != variables.end())
500  {
501  *iter = variables[*iter];
502  nb++;
503  }
504  */
505  for(var=variables.begin(); var != variables.end(); var++)
506  {
507  start = 0;
508  while ((start = iter->find(var->first, start)) != string::npos)

512  }
513  }

519  * @param scriptFileName the filename of the script to execute
520  * @param variables the variables of the script
521  * @param gOptions the global options of xmldiff
522  * @return status code :
523  * 0 no problems
524  * -10 file not found

526 int executeFile(string scriptFileName, const map<string, string> & variables, const struct globalOptions & gOptions)
527 {
528  int rc = 0;
529  struct globalOptions options = gOptions;
530  struct appCommand cmd;
531  ifstream fin;

534  vector<string> tokens;
535

537  fin.open(scriptFileName.c_str());
538  if (!fin.is_open()) throwError(XD_Exception::XDE_READ_ERROR, "Script file not found.");
539  while (fin.is_open() && (!fin.eof()))
540  {
541  fin.getline(cLine, sizeof(cLine)); line = cLine;
542  if (line == "") continue;
543  ((struct globalOptions &)cmd) = options;
544  tokens = tokenizeCommand(line);
545  replaceTokens(tokens, variables);
546  rc = parseCommandLine(tokens, cmd);
547  if (cmd.action == XD_OPTIONS)
548  {
549  options = (struct globalOptions) cmd;
550  // printConfiguration(options);
551  }
552  else
553  {
554  rc = executeAction(cmd);
555  }
556  }
557  flushXmlFiles(options);
558  return rc;
